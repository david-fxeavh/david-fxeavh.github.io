// JavaScript设备检测代码 function detectDevice() { // 方法1: 通过User Agent检测 const isMobileUA = /Android|webOS|iPhone|iPad|iPod|BlackBerry|IEMobile|Opera Mini/i.test(navigator.userAgent); // 方法2: 通过屏幕尺寸检测 const isMobileScreen = window.innerWidth <= 768; // 方法3: 通过触摸支持检测 const isTouchDevice = 'ontouchstart' in window || navigator.maxTouchPoints > 0; // 综合判断 return isMobileUA && isMobileScreen && isTouchDevice; } // 根据检测结果显示内容 if (detectDevice()) { document.getElementById('mobileContent').style.display = 'block'; document.getElementById('desktopContent').style.display = 'none'; } else { document.getElementById('mobileContent').style.display = 'none'; document.getElementById('desktopContent').style.display = 'block'; }

实现说明

此页面使用JavaScript检测设备类型，并根据检测结果动态显示不同的HTML内容。

检测方法包括：

  • User Agent检测: 检查浏览器用户代理字符串中的移动设备标识
  • 屏幕尺寸检测: 检查视口宽度是否小于等于768px
  • 触摸支持检测: 检查设备是否支持触摸操作

在实际项目中，建议结合多种检测方法以提高准确性。
